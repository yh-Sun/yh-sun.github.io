# 大象、吃象工具、水接口
interface Elephant { void eat();}
interface EatTool { void put();}
interface Water { void drink();}

# 中国象、筷、茶
class ChinaElephant implements Elephant {
  @Override public void eat() {sout.("中国象真香")}
}
class ChinaEatTool implements EatTool {
  @Override public void put() {sout.("夹")}
}
class ChinaWater implements Water {
  @Override public void drink() {sout.("品茶")}
}

# 工厂
public interface Factory {createElephant(); createEatTool(); createWater();}
public class ChinaFactory implements Factory {
  @Override createElephant(){new ChinaElephant()};
  @Override createEatTool(){new ChinaEatTool()};
  @Override createWater(){new ChinaWater()};
}

# 开吃
new ChinaFactory().createElephant().createEatTool().createWater();
