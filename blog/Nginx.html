# 安装nginx所需环境

rpm -Uvh ./*.rpm --nodeps --force

# 解压nginx

tar -zxvf nginx-1.12.2.tar.gz

# 安装nginx

cd nginx-1.12.2

配置
./configure --with-http_ssl_module
如果不是root用户没有usr等文件夹权限，需要使用prefix指定安装目录
./configure --with-http_ssl_module --prefix="~/nginx/conf" --with-http_stub_status_module

# 编译
make
# 安装
make install

# 配置环境变量量
vim /etc/profile
export NGINX_HOME=/usr/local/nginx
export PATH=$PATH:$NGINX_HOME/sbin
# 使环境变量量⽣生效
source /etc/profile

# 查看nginx版本
nginx -v

# 启动

~/nginx/conf-1.18/sbin/nginx -c ~/nginx/conf-1.18/conf/nginx.conf

启动时出现：nginx: [emerg] bind() to 0.0.0.0:80 failed (13: Permission denied)，原因是没有权限启动80端口（Linux只有root用户可以使用1024一下的端口）；

nginx: the configuration file /home/work/local-qc/nginx/conf/nginx.conf syntax is ok
nginx: [emerg] bind() to 0.0.0.0:80 failed (13: Permission denied)
nginx: configuration file /home/work/local-qc/nginx/conf/nginx.conf test failed
原因：当前用户对该位置没有写入权限
解决办法：
1.使用命令：sudo /usr/local/nginx/sbin/nginx -c /usr/local/nginx/conf/nginx.conf 以root权限启动
2.使用命令：sudo chmod -R a+rw /usr/local/nginx 给所有用户赋权限（个人学习，不考虑安全问题）
                    /usr/local/nginx/sbin/nginx -c /usr/local/nginx/conf/nginx.conf  启动Nginx

注：以非root权限启动时，会出现 nginx: [emerg] bind() to 0.0.0.0:80 failed (13: Permission denied) 错误
原因：Linux只有root用户可以使用1024一下的端口
解决办法：1.已root权限启动
  2.将 /usr/local/nginx/conf/nginx.conf 文件中的80端口改为1024以上
server {
# listen 80
   listen 8080
……

# nginx脚本安装

  • 安装包：nginx-1.18.0.tar.gz
  • 脚本：shell-nginx.sh
    • 将安装包与脚本放置于同级目录，执行sh脚本
    • 非root用户无法启动80端口，在此不使用sudo，直接修改默认端口为8080即可
    • 前端的配置文件[http_conf.d]与nginx配置[location proxy]目前为手动加入
    • 操作命令：
      • cd .../nginx-1.18.0
      • 启动：sbin/nginx -c conf/nginx.conf
      • 停止：sbin/nginx -s stop
      • 重启：sbin/nginx -s reload
#!/usr/bin/env bash

# nginx部署目标路径
NGINX_DIR="/home/xiandai"
# nginx安装包路径
DEPLOY_NGINX="nginx-1.18.0" 

# 将安装包解压到目标路径
function unGzip() {
    deploy_file=${DEPLOY_NGINX}.tar.gz
    cp ${deploy_file} ${NGINX_DIR}
    cd ${NGINX_DIR}
    
    tar -zxvf ${deploy_file}
    rm -rf ${deploy_file}

    info 'end un gzip and clear dir.'	
}

# 检测gcc、openssl、zlib、pcre是否安装
function pcre() {
    package="pcre"
    pName=$(rpm -qa | grep ${package})
    if [ $? -eq 0 ]
    then
        echo "软件包"${pName}"已经安装。"
    else
       echo "软件包"${package}"没有安装"
       # yum install pcre
    fi
}

# nginx安装配置
function nginx() {
    package="nginx"
    pName=$(rpm -qa | grep ${package})
    if [ $? -eq 0 ]
    then
       echo "软件包"${pName}"已经安装。"
    else
       echo "软件包"${package}"没有安装"
       cd ${NGINX_DIR}/${DEPLOY_NGINX}
	   
       # 安装文件的存放目录，configure的配置都放在target目录下（sbin、conf、html等）
       mkdir target
       ./configure --prefix=${NGINX_DIR}/${DEPLOY_NGINX}/target --with-http_ssl_module --with-http_stub_status_module
	   
	   # 编译
       make
       make install
       
       # 运行
       cd target
	   # cd /home/xiandai/nginx-1.18.0/target
	   # sbin/nginx -s stop
       sbin/nginx -c conf/nginx.conf
    fi
}

function main() {
    unGzip
    pcre
    nginx
}

main
