# 参考文献

https://www.cnblogs.com/yanjieli/p/10864648.html

  • Saltstack原理
Salt使用server-agent通信模型，服务端组件被称为Salt master，agent被称为Salt minion
Salt master主要负责向Salt minions发送命令，然后聚合并显示这些命令的结果。一个Salt master可以管理多个minion系统
Salt server与Salt minion通信的连接由Salt minion发起，这也意味着Salt minion上不需要打开任何传入端口（从而减少攻击）。Salt server使用端口4505和4506，必须打开端口才能接收到访问连接

  • （端口4505）所有Salt minions都需要建立一个持续连接到他们收听消息的发布者端口。命令是通过此端口异步发送给所有连接，这使命令可以在大量系统上同时执行。
  • Server （端口4506）Salt minios根据需要连接到请求服务器，将结果发送给Salt master，并安全地获取请求的文件或特定minion相关的数据值（称为Salt pillar）。连接到这个端口的连接在Salt master和Salt minion之间是1:1（不是异步）。

# yum安装salt-Master、salt-Minion

  1. 配置yum仓库
# 使用阿里云yum
[root@master ~]# yum -y install https://mirrors.aliyun.com/saltstack/yum/redhat/salt-repo-latest-2.el7.noarch.rpm
[root@master ~]# sed -i "s/repo.saltstack.com/mirrors.aliyun.com\/saltstack/g" /etc/yum.repos.d/salt-latest.repo
# clean cache
[root@master ~]# yum clean all
[root@master ~]# yum makecache
  1. 安装salt-Master
[root@master ~]# yum -y install salt-master
# 设置开启自启，并启动salt-master
[root@master ~]# systemctl enable salt-master
[root@master ~]# systemctl start salt-master
  1. 安装salt-Minion
[root@minion ~]# yum -y install salt-minion

# 指向salt-master网络地址，可以是主机名，也可以是IP地址；此处我们使用ip地址
# 首先备份一下minion，语法详见<附录1>
[root@minion ~]# cp /etc/salt/minion{,.back}
# salt-master：master ip，也可以直接手动vim
[root@minion ~]# sed -i '/#master: /c\master: salt-master' /etc/salt/minion

# 设置开启自启，并启动salt-minion
[root@minion ~]# systemctl enable salt-master
[root@minion ~]# systemctl start salt-master
  1. 认证
    1. 未认证pki
[root@master ~]$ tree /etc/salt/pki
/etc/salt/pki
├── master
│   ├── master.pem
│   ├── master.pub
│   ├── minions
│   ├── minions_autosign
│   ├── minions_denied
│   ├── minions_pre
│   │   └── VM-0-9-centos，为minion配置的主机名称
│   └── minions_rejected
└── minion

[root@minion ~]$ tree /etc/salt/pki/
/etc/salt/pki/
├── master
└── minion
    ├── minion.pem
    └── minion.pub
  1. 已认证pki
[root@master ~]$ tree /etc/salt/pki
/etc/salt/pki
├── master
│   ├── master.pem
│   ├── master.pub
│   ├── minions
│   │   └── VM-0-9-centos，为minion配置的主机名称
│   ├── minions_autosign
│   ├── minions_denied
│   ├── minions_pre
│   └── minions_rejected
└── minion

[root@minion ~]$ tree /etc/salt/pki/
/etc/salt/pki/
├── master
└── minion
    ├── minion_master.pub
    ├── minion.pem
    └── minion.pub

以上两段对比可以看出，当认证通过时，master pki的minion将由minions_pre流转至minions，minion的pki将多出master的公钥Public Key

  1. 如何认证（salt-key）
# 配置master自动接受请求认证(master上配置 /etc/salt/master)，在master启动前就可配置，那就无需做认证的操作了
auto_accept: True

# 当然，也可以手动接受指定的minion
[root@master ~]# salt-key -L 
Accepted Keys:        #已经接受的key
Denied Keys:          #拒绝的key
Unaccepted Keys:      #未加入的key
Rejected Keys:        #吊销的key

# 常用参数
-L  #查看KEY状态
-A  #允许所有
-D  #删除所有
-a  #认证指定的key
-d  #删除指定的key
-r  #注销掉指定key（该状态为未被认证）

# 添加指定minion的key
[root@master ~]# salt-key -a salt-minion01 -y
The following keys are going to be accepted:
Unaccepted Keys:
salt-minion01
Key for minion salt-minion01 accepted.

#添加所有minion的key
[root@master ~]# salt-key -A -y

# 让我们看看认证通过的样子吧~
[root@master ~]# salt-key -L 
Accepted Keys:
salt-minion01
Denied Keys:
Unaccepted Keys:
Rejected Keys:

# sls配置使用

# 定义环境目录

file_roots:
  base:
    - /srv/salt/base
  dev:
    - /srv/salt/dev
  prod:
    - /srv/salt/prod

初始情况下，是没有以上目录的，需要自己手动创建

mkdir -p /srv/salt/{base,dev,prod}

# 重启master

# sls

本次演示以默认base环境为准，如果采用其他环境，如prod，只需执行sls时指定“saltenv=dev”

# 创建sls脚本、资源文件夹files
[root@master ~]# cd /srv/salt/base

# 推送文件file.managend.sls
[root@master ~]# touch salt-managend.sls
~/test_salt.log:
  file.managed:
    # 如果不指定name，可以将那么的值声明在该命令块第一行
    # - name: ~/test_salt.log
    - source: salt://files/test.log
    - user: root
    - group: root
    - mode: 644
	
# 执行sls脚本
[root@master ~]# salt '*' state.sls salt-managend.sls



# 在目标服务器执行脚本smd.script.sls
[root@master ~]# touch salt-script.sls
test_sh:
  cmd.script:
    - source: salt://files/test.sh
    - user: root
	
# 执行sls脚本
[root@master ~]# salt '*' state.sls salt-script.sls
