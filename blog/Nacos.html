# nacos经常输出心跳日志，将其级别升高不打印的方式

# 1、根据心跳日志，定位日志输出的包名

c.a.n.client.config.impl.ClientWorker : get changedGroupKeys:[] c.a.n.client.config.impl.ClientWorker : get changedGroupKeys:[] c.a.n.client.config.impl.ClientWorker : get changedGroupKeys:[] c.a.n.client.config.impl.ClientWorker : get changedGroupKeys:[]

# 2、在 IDEA 中搜索 ClientWorker 所在的包路径

package com.alibaba.nacos.client.config.impl;

# 3、在任意格式的配置文件中将该包路径的日志设置为 ERROR 或者 WARN 级别即可

# Nacos 注册中心客户端心跳日志禁用 get changedGroupKeys:[] 刷屏

logging: level: com.alibaba.nacos.client.config.impl: WARN

# 4、如果是 Spring Cloud Gateway 则需要配置为

logging: level: com.alibaba.nacos.client.*: WARN

# nacos脚本安装

  • 安装包：nacos-server-2.0.1.tar.gz
  • 脚本：shell-nacos.sh
    • 将安装包与脚本放置于同级目录，执行sh脚本
#!/usr/bin/env bash

# nacos安装到的目标目录
NACOS_DEPLOY_DIR="/home/xiandai"

# nacos文件目录，用于tar、cp、mv指定文件名
DEPLOY_GZ_FILE_NAME="nacos-server-2.0.1"

# nacos集群IP（本次并未配置集群）
IP_1="192.168.0.1"
IP_2="192.168.0.2"
IP_3="192.168.0.3"

# 数据库配置
spring.datasource.platform=mysql
db.num=1
db.url.0=jdbc:mysql://10.1.1.100:3306/kscd_dev?characterEncoding=utf8&connectTimeout=1000&socketTimeout=3000&autoReconnect=true&useUnicode=true&useSSL=false&serverTimezone=UTC
db.user=kscd_dev_user
db.password=bql^2xP1EriKqTQm

# 将nacos安装包解压到目标目录
function unGzip() {
    deploy_file=${DEPLOY_GZ_FILE_NAME}.tar.gz
    cp ${deploy_file} ${NACOS_DEPLOY_DIR}

    cd ${NACOS_DEPLOY_DIR}
    
    tar -zxvf ${deploy_file}
    rm -rf  ${deploy_file}

    info 'end un gzip and clear dir.'
}

# 集群配置、将数据库配置加入nacos配置文件中
function configNacosClusterIp() {
    nacos_config_dir=${NACOS_DEPLOY_DIR}/nacos/conf
    cd ${nacos_config_dir}
	
    # touch cluster.conf
    # echo '# nacos cluster ip'
    # echo "${IP_1}:8848" >> cluster.conf
    # echo "${IP_2}:8848" >> cluster.conf
    # echo "${IP_3}:8848" >> cluster.conf
    # cat cluster.conf
    # info "end nacos cluster ip config: ${nacos_config_dir}/cluster.conf"

    echo "spring.datasource.platform=mysql" >> application.properties
    echo "db.num=1" >> application.properties
    echo "db.url.0=${DB_URL}" >> application.properties
    echo "db.user=${DB_USER}" >> application.properties
    echo "db.password=${DB_PASSWORD}" >> application.properties

    warn "please check ${nacos_config_dir}/application.properties db config!"
}

# 启动nacos（本次采用单机启动）
function start() {
    nacos_bin_dir=${NACOS_DEPLOY_DIR}/nacos/bin
    cd ${nacos_bin_dir}

    # 单机启动
    sh startup.sh -m standalone &
    ps -ef|grep nacos
}


function main() {
    unGzip
    configNacosClusterIp
    start
}

main
